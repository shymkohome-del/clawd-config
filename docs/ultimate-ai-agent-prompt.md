     /// 
     /// Throws [ServiceInitializationException] if dependencies
     /// cannot be initialized properly.
     CryptoPriceAlertService({

       required NotificationService notificationService,
       Logger? logger,
     }) : _priceRepository = priceRepository,
          _notificationService = notificationService,
          _logger = logger ?? Logger('CryptoPriceAlertService') {
       _initializeService();
     }

       try {

         await _notificationService.initialize();
         _logger.info('CryptoPriceAlertService initialized successfully');
       } catch (e, stackTrace) {
         _logger.severe('Failed to initialize service', e, stackTrace);
         throw ServiceInitializationException(
           'Unable to initialize price alert service: ${e.toString()}',
           originalError: e,
         );
       }
     }
   }
